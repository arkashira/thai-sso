  public readonly code: string;
  public readonly status: number;

  constructor(code: string, message: string, status = 500) {
    super(message);
    this.name = 'PdfServiceError';
    this.code = code;
    this.status = status;
  }
}

/**
 * Generates a simple PDF containing the provided text.
 * Works in Node (server) and in the browser.
 *
 * @param content Text to embed in the PDF.
 * @returns Uint8Array of the PDF binary.
 */
export async function generatePdf(content: string): Promise<Uint8Array> {
  const pdfDoc = await PDFDocument.create();
  const page = pdfDoc.addPage();
  const font = await pdfDoc.embedFont(StandardFonts.Helvetica);

  const { width, height } = page.getSize();
  const fontSize = 12;
  const textWidth = font.widthOfTextAtSize(content, fontSize);
  const textHeight = font.heightAtSize(fontSize);

  page.drawText(content, {
    x: (width - textWidth) / 2,
    y: (height - textHeight) / 2,
    size: fontSize,
    font,
    color: rgb(0, 0, 0),
  });

  return await pdfDoc.save();
}
