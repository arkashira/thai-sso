const winston = require('winston');

const logger = winston.createLogger({
  level: 'info',

  transports: [

    new winston.transports.File({ filename: 'combined.log' }),
  ],

const userStatisticsSchema = Joi.object({
  totalUsers: Joi.number().required(),
  activeUsers: Joi.number().required(),
  lastLoginDates: Joi.array().required(),
});

router.get('/user-statistics', async (req, res) => {
  try {
    const users = await User.findAll();
    const statistics = users.reduce((acc, user) => {
      acc.totalUsers += 1;
      acc.activeUsers += 1;
      acc.lastLoginDates.push(user.lastLogin);
      return acc;
    }, { totalUsers: 0, activeUsers: 0, lastLoginDates: [] });
    const response = { totalUsers: statistics.totalUsers, activeUsers: statistics.activeUsers, lastLoginDates: statistics.lastLoginDates };
    await userStatisticsSchema.validateAsync(response);
    res.json(response);
  } catch (error) {
    logger.error(error);
    res.status(500).json({ message: 'Error fetching user statistics' });
  }
});

module.exports = router;
