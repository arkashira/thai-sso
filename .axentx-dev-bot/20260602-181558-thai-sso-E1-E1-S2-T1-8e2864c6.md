```js
// src/services/jwt.js
const jwt = require('jsonwebtoken');

const JWT_SECRET = process.env.JWT_SECRET;
if (!JWT_SECRET) {
  throw new Error('JWT_SECRET must be set in the environment');
}

/**
 * Create a signed JWT containing the user id.
 * @param {number|string} userId
 * @returns {string} signed token
 */
function createToken(userId) {
  const payload = { sub: userId };
  const options = { expiresIn: '1h' }; // 1‑hour expiry
  return jwt.sign(payload, JWT_SECRET, options);
}

/**
 * Verify a JWT and return the decoded payload.
 * @param {string} token
 * @returns {object} decoded payload
 * @throws {Error} if token is invalid or expired
 */
function verifyToken(token) {
  return jwt.verify(token, JWT_SECRET);
}

module.exports = { createToken, verifyToken };
```

### 2. `src/api/routes/auth.js`

```js
// src/api/routes/auth.js
const express = require('express');
const bcrypt = require('bcryptjs');
const router = express.Router();
const { createToken } = require('../../services/jwt');
const { getUserByUsername } = require('../../services/user'); // assume exists
const knex = require('../../db'); // Knex instance

/**
 * POST /login
 * Body: { username, password }
 * On success: sets HttpOnly cookie `token`, logs audit, returns 200.
 * On failure: returns 400/401.
 */
router.post('/login', async (req, res) => {
  const { username, password } = req.body;

  if (!username || !password) {
    return res.status(400).json({ error: 'Missing credentials' });
  }

  try {
    const user = await getUserByUsername(username);
    if (!user) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }

    const matches = await bcrypt.compare(password, user.password_hash);
    if (!matches) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }

    const token = createToken(user.id);

    // Set HttpOnly cookie
    res.cookie('token', token, {
      httpOnly: true,
      sameSite: 'lax',
      secure: process.env.NODE_ENV === 'production',
      maxAge: 60 * 60 * 1000, // 1 hour
    });

    // Audit log
    const ip =
      req.headers['x-forwarded-for'] ||
      req.connection.remoteAddress ||
      req.ip;
    await knex('audit_sessions').insert({
      user_id: user.id,
      ip_address: ip,
      login_time: knex.fn.now(),
    });

    return res.status(200).json({ message: 'Login successful' });
  } catch (err) {
    console.error('Login error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
});

module.exports = router;
```

```javascript
// jwt.test.js
const jwt = require('jsonwebtoken');
const { generateToken, verifyToken } = require('../../src/services/jwt');

const SECRET = process.env.JWT_SECRET || 'test-secret';

describe('JWT Service', () => {
  describe('generateToken', () => {
    test('returns a non-empty string', () => {
      const token = generateToken({ userId: 1 });
      expect(typeof token).toBe('string');
      expect(token.length).toBeGreaterThan(0);
    });

    test('payload contains userId and iat', () => {
      const token = generateToken({ userId: 42 });
      const decoded = jwt.decode(token);
      expect(decoded.userId).toBe(42);
      expect(decoded.iat).toBeDefined();
    });

    test('exp is exactly iat + 3600 seconds', () => {
      const before = Math.floor(Date.now() / 1000);
      const token = generateToken({ userId: 7 });
      const decoded = jwt.decode(token);
      expect(decoded.exp).toBe(decoded.iat + 3600);
      expect(decoded.exp).toBeGreaterThanOrEqual(before + 3600);
    });

    test('token is signed with the configured secret', () => {
      const token = generateToken({ userId: 1 });
      expect(() => jwt.verify(token, SECRET)).not.toThrow();
    });

    test('token signed with wrong secret fails verification', () => {
      const token = generateToken({ userId: 1 });
      expect(() => jwt.verify(token, 'wrong-secret')).toThrow();
    });
  });

  describe('verifyToken', () => {
    test('returns decoded payload for a valid token', () => {
      const token = generateToken({ userId: 99, role: 'admin' });
      const decoded = verifyToken(token);
      expect(decoded).not.toBeNull();
      expect(decoded.userId).toBe(99);
    });

    test('throws error for expired token', () => {
      const expiredToken = jwt.sign({ userId: 1, exp: Math.floor(Date.now() / 1000) - 1 }, SECRET);
      expect(() => verifyToken(expiredToken)).toThrow();
    });

    test('throws error for tampered token', () => {
      const token = generateToken({ userId: 1 });
      const tamperedToken = token + 'tampered';
      expect(() => verifyToken(tamperedToken)).toThrow();
    });
  });
});
```
