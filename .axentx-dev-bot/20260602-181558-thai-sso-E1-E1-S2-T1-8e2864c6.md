### 1. `src/services/jwt.js`

// src/services/jwt.js
const jwt = require('jsonwebtoken');

const JWT_SECRET = process.env.JWT_SECRET;
if (!JWT_SECRET) {
  throw new Error('JWT_SECRET must be set in the environment');
}

/**
 * Create a signed JWT containing the user id.
 * @param {number|string} userId
 * @returns {string} signed token
 */
function createToken(userId) {
  const payload = { sub: userId };
  const options = { expiresIn: '1h' }; // 1‑hour expiry
  return jwt.sign(payload, JWT_SECRET, options);
}

/**
 * Verify a JWT and return the decoded payload.
 * @param {string} token
 * @returns {object} decoded payload
 * @throws {Error} if token is invalid or expired
 */
function verifyToken(token) {
  return jwt.verify(token, JWT_SECRET);
}

module.exports = { createToken, verifyToken };
```

### 2. `src/api/routes/auth.js`

```js
// src/api/routes/auth.js
const express = require('express');
const bcrypt = require('bcryptjs');
const router = express.Router();
const { createToken } = require('../../services/jwt');
const { getUserByUsername } = require('../../services/user'); // assume exists
const knex = require('../../db'); // Knex instance

/**
 * POST /login
 * Body: { username, password }
 * On success: sets HttpOnly cookie `token`, logs audit, returns 200.
 * On failure: returns 400/401.
 */
router.post('/login', async (req, res) => {
  const { username, password } = req.body;

  if (!username || !password) {
    return res.status(400).json({ error: 'Missing credentials' });
  }

  try {
    const user = await getUserByUsername(username);
    if (!user) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }

    const matches = await bcrypt.compare(password, user.password_hash);
    if (!matches) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }

    const token = createToken(user.id);

    // Set HttpOnly cookie
    res.cookie('token', token, {
      httpOnly: true,
      sameSite: 'lax',
      secure: process.env.NODE_ENV === 'production',
      maxAge: 60 * 60 * 1000, // 1 hour
    });

    // Audit log
    const ip =
      req.headers['x-forwarded-for'] ||
      req.connection.remoteAddress ||
      req.ip;
    await knex('audit_sessions').insert({
      user_id: user.id,
      ip_address: ip,
      login_time: knex.fn.now(),
    });

    return res.status(200).json({ message: 'Login successful' });
  } catch (err) {
    console.error('Login error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
});

module.exports = router;
```
