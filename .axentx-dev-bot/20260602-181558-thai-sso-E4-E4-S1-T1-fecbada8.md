```jsx
import React, { useEffect, useState } from 'react';
import { getUserStats } from '../api/userStats';

const UserStatsCard = () => {
  const [stats, setStats] = useState({
    users: [],
    activeUsers: 0,
    lastLoginDates: {}
  });
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchUserStats = async () => {
      try {
        const data = await getUserStats();
        setStats(data);
      } catch (error) {
        // Handle error silently in UI or show error state
      } finally {
        setLoading(false);
      }
    };

    // Fetch immediately on mount
    fetchUserStats();

    // Set up polling interval (5 minutes = 300,000 ms)
    const intervalId = setInterval(fetchUserStats, 300000);

    // Cleanup interval on unmount
    return () => clearInterval(intervalId);
  }, []);

  if (loading) return <div>Loading user statistics...</div>;

  const { users, activeUsers, lastLoginDates } = stats;

  return (
    <div className="user-stats-card">
      <h2>User Statistics</h2>
      
      <div className="stats-summary">
        <p><strong>Total Users:</strong> {users.length}</p>
        <p><strong>Active Users:</strong> {activeUsers}</p>
      </div>

      <div className="user-list">
        <h3>Last Logins</h3>
        {users.map((user) => (
          <div key={user.id}>
            <p>{user.username}: {new Date(user.lastLogin).toLocaleString()}</p>
          </div>
        ))}
      </div>
    </div>
  );
};

export default UserStatsCard;
```

// UserStatsCard.jsx
import React, { useState, useEffect } from 'react';
import getUserStats from '../api/userStats';

const UserStatsCard = () => {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [activeUsers, setActiveUsers] = useState(0);
  const [lastLoginDates, setLastLoginDates] = useState([]);

  useEffect(() => {
    const intervalId = setInterval(async () => {
      const newUsers = await getUserStats();
      if (newUsers) {
        setUsers(newUsers);
        setActiveUsers(newUsers.activeUsers);
        setLastLoginDates(newUsers.users.map(user => user.lastLoginDate));
        setLoading(false);
      }
    }, 300000);
    return () => clearInterval(intervalId);
  }, []);

  if (loading) {
    return <div>Loading...</div>;
  }

  return (
    <div>
      <h2>Total Users: {users.length}</h2>
      <p>Active Users: {activeUsers}</p>
      <h3>Last Login Dates:</h3>
      <ul>
        {lastLoginDates.map((date, index) => (
          <li key={index}>{date}</li>
        ))}
      </ul>
    </div>
  );
};

export default UserStatsCard;
