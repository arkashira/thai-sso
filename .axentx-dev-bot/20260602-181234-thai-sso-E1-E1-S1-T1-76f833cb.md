+"""
+SSO Router
+-----------
+Implements a minimal OpenID‑Connect *Authorization Code* flow against the
+Thai government identity provider (https://oidc.thai.gov).  The flow is:
+
+1. **GET /sso/login** – redirects the user‑agent to the provider’s
+   authorization endpoint with the required query parameters.
+2. **GET /sso/callback** – receives the ``code`` query param, exchanges it
+   for an ``id_token`` (JWT) and stores the token in the session store.
+
+All secrets (client_id, client_secret, redirect_uri) are read from environment
+variables so the code works both locally and in production.  The implementation
+is deliberately lightweight – it uses ``httpx`` for the token‑exchange request
+and the existing ``session`` helper to persist the user information.
+"""
+
+import os
+import urllib.parse
+import base64
+import json
+from typing import Dict
+
+import httpx
+from fastapi import APIRouter, Request, Response, HTTPException, status
+from fastapi.responses import RedirectResponse
+
+from app.session import store_user_session
+
+router = APIRouter(prefix="/sso", tags=["sso"])
+
+# --------------------------------------------------------------------------- #
+# Configuration – read once at import time
+# --------------------------------------------------------------------------- #
+CLIENT_ID = os.getenv("THAI_SSO_CLIENT_ID", "demo-client-id")
+CLIENT_SECRET = os.getenv("THAI_SSO_CLIENT_SECRET", "demo-client-secret")
+PROVIDER_AUTH_ENDPOINT = os.getenv(
+    "THAI_SSO_AUTH_ENDPOINT",
+    "https://oidc.thai.gov/authorize",
+)
+PROVIDER_TOKEN_ENDPOINT = os.getenv(
+    "THAI_SSO_TOKEN_ENDPOINT",
+    "https://oidc.thai.gov/token",
+)
+REDIRECT_URI = os.getenv(
+    "THAI_SSO_REDIRECT_URI",
+    "http://localhost:8000/sso/callback",
+)
+SCOPE = "openid profile email"
+
+
+def _build_auth_url(state: str) -> str:
+    """Construct the provider‑authorization URL."""
+    query = {
+        "response_type": "code",
+        "client_id": CLIENT_ID,
+        "redirect_uri": REDIRECT_URI,
+        "scope": SCOPE,
+        "state": state,
+    }
+    return f"{PROVIDER_AUTH_ENDPOINT}?{urllib.parse.urlencode(query)}"
