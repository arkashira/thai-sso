// src/utils/email.js
const nodemailer = require('nodemailer');

/**
 * Create a Nodemailer transport that works in prod *and* in local dev.
 *
 * - In production you must set the following env vars:
 *   SMTP_HOST, SMTP_PORT, SMTP_SECURE, SMTP_USER, SMTP_PASS
 * - In local dev you can skip all of them and the transport will
 *   point to a local test server (e.g. smtp4dev on port 1025).
 *
 * The function is idempotent – calling it multiple times will reuse
 * the same transport instance.
 */
let _transporter = null;

function getTransporter() {
  if (_transporter) return _transporter;

  const host = process.env.SMTP_HOST || 'localhost';
  const port = process.env.SMTP_PORT ? parseInt(process.env.SMTP_PORT, 10) : 1025;
  const secure = process.env.SMTP_SECURE === 'true'; // 465 = true, others = false

  const auth = process.env.SMTP_USER
    ? {
        user: process.env.SMTP_USER,
        pass: process.env.SMTP_PASS || '',
      }
    : undefined; // no auth for local test server

  _transporter = nodemailer.createTransport({
    host,
    port,
    secure,
    auth,
  });

  return _transporter;
}

/**
 * Send an e‑mail using the configured transporter.
 *
 * @param {Object} mailOptions – nodemailer options (to, from, subject, etc.)
 * @returns {Promise<SentMessageInfo>}
 */
async function sendMail(mailOptions) {
  const transporter = getTransporter();
  return transporter.sendMail(mailOptions);
}

module.exports = { sendMail };
```

**Why this works**

* **Secrets are never in source control** – they come from `process.env`.  
* **Local dev is painless** – if no env vars are set, the code falls back to a local SMTP server on port 1025.  
* **Single transport instance** – avoids creating a new connection for every mail.

### 2. `src/utils/pdf.js`

// src/utils/pdf.js
const { spawnSync } = require('child_process');
const fs = require('fs');
const path = require('path');

/**
 * Generate a PDF from an HTML file using wkhtmltopdf.
 *
 * @param {string} htmlFile - Path to the source HTML file.
 * @param {string} pdfFile  - Path where the PDF should be written.
 * @returns {Promise<void>}
 *
 * @throws {Error} If the binary is missing, the input file is missing,
 *                 or wkhtmltopdf exits with a non‑zero status.
 */
async function generatePdf(htmlFile, pdfFile) {
  // Resolve the binary – allow override via env var (CI, Docker, etc.)
  const wkhtmltopdfPath = process.env.WKHTMLTOPDF_PATH || 'wkhtmltopdf';

  // Validate input file
  if (!fs.existsSync(htmlFile)) {
    throw new Error(`HTML source not found: ${htmlFile}`);
  }

  // Ensure output directory exists
  const outDir = path.dirname(pdfFile);
  if (!fs.existsSync(outDir)) {
    fs.mkdirSync(outDir, { recursive: true });
  }

  // Run the binary synchronously – we want to surface errors immediately.
  const result = spawnSync(wkhtmltopdfPath, [htmlFile, pdfFile], {
    encoding: 'utf-8',
    stdio: ['ignore', 'pipe', 'pipe'],
  });

  if (result.status !== 0) {
    const errMsg = result.stderr || 'Unknown error';
    throw new Error(`wkhtmltopdf failed (${result.status}): ${errMsg}`);
  }

  // Double‑check that the file was written
  await new Promise((resolve, reject) => {
    fs.access(pdfFile, fs.constants.F_OK, (err) => {
      if (err) return reject(new Error(`PDF not created: ${pdfFile}`));
      resolve();
    });
  });
}

module.exports = { generatePdf };
