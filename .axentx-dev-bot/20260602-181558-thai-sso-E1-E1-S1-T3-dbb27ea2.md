import os
from dataclasses import dataclass
from dotenv import load_dotenv

# Load .env if present (development); in production the env will already be set
load_dotenv()

@dataclass(frozen=True)
class SMTPConfig:
    host: str
    port: int
    email: str
    password: str
    use_tls: bool

def _get_bool(value: str | None) -> bool:
    return str(value).lower() in {"1", "true", "yes", "on"}

def load_smtp_config() -> SMTPConfig:
    """Read required SMTP settings from the environment and raise if missing."""
    host = os.getenv("SMTP_HOST")
    port = os.getenv("SMTP_PORT")
    email = os.getenv("SMTP_EMAIL")
    password = os.getenv("SMTP_PASSWORD")
    use_tls = _get_bool(os.getenv("SMTP_USE_TLS", "true"))

    missing = [name for name, val in [
        ("SMTP_HOST", host),
        ("SMTP_PORT", port),
        ("SMTP_EMAIL", email),
        ("SMTP_PASSWORD", password),
    ] if not val]

    if missing:
        raise EnvironmentError(
            f"Missing required SMTP environment variables: {', '.join(missing)}"
        )

    try:
        port_int = int(port)
    except ValueError as exc:
        raise EnvironmentError("SMTP_PORT must be an integer") from exc

    return SMTPConfig(
        host=host,
        port=port_int,
        email=email,
        password=password,
        use_tls=use_tls,
    )

**`mailer.py`** – Minimal, testable SMTP wrapper.

```python
import smtplib
from email.message import EmailMessage
from config import load_smtp_config, SMTPConfig

class Mailer:
    def __init__(self, cfg: SMTPConfig | None = None):
        self.cfg = cfg or load_smtp_config()

    def send(self, to: str, subject: str, body: str) -> None:
        msg = EmailMessage()
        msg["From"] = self.cfg.email
        msg["To"] = to
        msg["Subject"] = subject
        msg.set_content(body)

        with smtplib.SMTP(self.cfg.host, self.cfg.port) as server:
            if self.cfg.use_tls:
                server.starttls()
            server.login(self.cfg.email, self.cfg.password)
            server.send_message(msg)
```
