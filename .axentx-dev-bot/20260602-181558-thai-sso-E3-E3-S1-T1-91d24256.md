```python
# test_docker_compose_structure.py
import yaml
import re

def load_compose():
    with open('docker-compose.yml') as f:
        return yaml.safe_load(f)

def test_top_level_keys():
    compose = load_compose()
    assert 'services' in compose
    assert isinstance(compose['services'], dict)

def test_service_names():
    compose = load_compose()
    services = compose['services']
    expected = {'postgres', 'redis', 'sso-api'}
    assert set(services.keys()) == expected

def test_postgres_image_and_env():
    svc = load_compose()['services']['postgres']
    assert re.match(r'^postgres:\d+.*-alpine$', svc['image'])
    env = svc.get('environment', {})
    for var in ['POSTGRES_DB', 'POSTGRES_USER', 'POSTGRES_PASSWORD']:
        assert var in env
        assert env[var].startswith('${') and env[var].endswith('}')

def test_redis_image_and_env():
    svc = load_compose()['services']['redis']
    assert re.match(r'^redis:\d+.*-alpine$', svc['image'])
    env = svc.get('environment', {})
    assert 'REDIS_PASSWORD' in env
    assert env['REDIS_PASSWORD'].startswith('${') and env['REDIS_PASSWORD'].endswith('}')

def test_sso_api_dependencies_and_ports():
    svc = load_compose()['services']['sso-api']
    assert svc['depends_on'] == ['postgres', 'redis']
    ports = svc.get('ports', [])
    assert '8080:8080' in ports
    # ensure all PDPA vars are referenced
    pdpa_vars = [v for v in svc.get('environment', {}) if v.startswith('PDPA_')]
    assert len(pdpa_vars) >= 1   # at least one PDPA setting required
```
