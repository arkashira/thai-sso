const auditMiddleware = require('../src/middleware/audit');
const { Audit } = require('../src/models/audit');
const httpMocks = require('node-mocks-http');

jest.mock('../src/models/audit'); // mock DB layer

describe('audit middleware', () => {
  beforeEach(() => jest.clearAllMocks());

  test('creates audit record on authenticated request', async () => {
    const req = httpMocks.createRequest({
      method: 'GET',
      url: '/dashboard',
      headers: { 'x-forwarded-for': '203.0.113.5' },
      user: { id: 42 } // set by JWT middleware
    });
    const res = httpMocks.createResponse();
    const next = jest.fn();

    await auditMiddleware(req, res, next);

    expect(Audit.create).toHaveBeenCalledWith(
      expect.objectContaining({
        user_id: 42,
        action: 'GET /dashboard',
        ip_address: '203.0.113.5'
      })
    );
    expect(next).toHaveBeenCalled();
  });

  test('does not audit unauthenticated request', async () => {
    const req = httpMocks.createRequest({ method: 'POST', url: '/login' });
    const res = httpMocks.createResponse();
    const next = jest.fn();

    await auditMiddleware(req, res, next);

    expect(Audit.create).not.toHaveBeenCalled();
    expect(next).toHaveBeenCalled();

  test('captures IP from request.ip when X-Forwarded-For absent', async () => {

    );
