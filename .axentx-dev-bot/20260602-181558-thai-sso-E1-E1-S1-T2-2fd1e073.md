## 1. Acceptance Criteria
1. **Password hashing** – When a new user is created, the plaintext password is never stored; the `password_hash` column contains a bcrypt hash (cost factor ≥ 12).  
2. **Hash verification** – The `User.verifyPassword(plain)` method returns `true` for the original password and `false` for any other string.  
3. **Deterministic API** – `hashPassword(plain)` returns a string that matches the bcrypt format (`$2b$<cost>$<22‑char‑salt><31‑char‑hash>`).  
4. **Performance** – Hashing a password ≤ 12 ms on the CI runner (node v20, typical CI hardware).  
5. **Error handling** – Supplying a non‑string or empty password throws a `TypeError` with a clear message; the user creation flow aborts and does not write a record.  
6. **Database integrity** – After successful registration, the `users` table contains a row with the correct email and a non‑null `password_hash`.

// __tests__/hash.test.js
const { hashPassword, comparePassword } = require('../src/utils/hash');

describe('hashPassword()', () => {
  test('returns a bcrypt hash string with correct format', async () => {
    const hash = await hashPassword('Secret123!');
    expect(typeof hash).toBe('string');
    expect(hash).toMatch(/^\$2[aby]\$\d{2}\$[./A-Za-z0-9]{53}$/);
  });

  test('uses a cost factor of at least 12', async () => {
    const hash = await hashPassword('Secret123!');
    const cost = parseInt(hash.split('$')[2], 10);
    expect(cost).toBeGreaterThanOrEqual(12);
  });

  test('throws TypeError for non‑string input', async () => {
    await expect(hashPassword(12345)).rejects.toThrow(TypeError);
    await expect(hashPassword('')).rejects.toThrow(/cannot be empty/);
  });
});

describe('comparePassword()', () => {
  const plain = 'My$tr0ngP@ss';
  let hash;

  beforeAll(async () => {
    hash = await hashPassword(plain);
  });

  test('returns true for matching password', async () => {
    const ok = await comparePassword(plain, hash);
    expect(ok).toBe(true);
  });

  test('returns false for non‑matching password', async () => {
    const ok = await comparePassword('WrongPass', hash);
    expect(ok).toBe(false);
  });
});
