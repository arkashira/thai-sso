PASS: Test plan for **thai-sso – password hashing with bcrypt** (implementation located in `src/utils/hash.js` and `src/models/user.js`).

---

## 1. Acceptance Criteria
1. **Hash Generation** – `hashPassword(plain)` returns a bcrypt hash string that matches the regex `^\$2[aby]\$\d{2}\$[./A-Za-z0-9]{53}$` and uses a cost factor **≥ 12**.
2. **No Plaintext Storage** – After a successful registration, the `users.password_hash` column contains a bcrypt hash; the plaintext password is never persisted.
3. **Verification Method** – `User.verifyPassword(plain)` (or `comparePassword(plain, hash)`) returns `true` for the original password and `false` for any other input.
4. **Error Handling** – Supplying a non‑string, `null`, `undefined`, or empty string to `hashPassword` throws a `TypeError` with a clear message; the registration flow aborts and no DB row is created.
5. **Performance** – Hashing a typical password (8‑32 characters) completes in **≤ 12 ms** on the CI runner (Node v20, 2 vCPU, 4 GB RAM). Measured over 10 runs, the average must be ≤ 12 ms.
6. **Database Integrity** – After registration, a row exists with:
   - `email` equal to the submitted address (validated format).
   - `password_hash` non‑null, matching the bcrypt format.
   - `created_at` timestamp set.
7. **Email Verification Trigger** – The registration flow calls `sendVerificationEmail(email)` **only after** the user record is successfully written; failures in hashing must prevent the email from being sent.

// __tests__/hash.test.js
const { hashPassword, comparePassword } = require('../src/utils/hash');

describe('hashPassword()', () => {
  test('returns a bcrypt hash string with correct format', async () => {
    const hash = await hashPassword('Secret123!');
    expect(typeof hash).toBe('string');
    expect(hash).toMatch(/^\$2[aby]\$\d{2}\$[./A-Za-z0-9]{53}$/);
  });

  test('uses a cost factor of at least 12', async () => {
    const hash = await hashPassword('Secret123!');
    const cost = Number(hash.split('$')[2]);
    expect(cost).toBeGreaterThanOrEqual(12);
  });

  test('hashes different passwords to different hashes (salted)', async () => {
    const h1 = await hashPassword('PasswordOne');
    const h2 = await hashPassword('PasswordOne');
    expect(h1).not.toBe(h2);
  });

  test('throws TypeError for non‑string input', async () => {
    await expect(hashPassword(12345)).rejects.toThrow(TypeError);
    await expect(hashPassword(null)).rejects.toThrow(TypeError);
    await expect(hashPassword(undefined)).rejects.toThrow(TypeError);
    await expect(hashPassword('')).rejects.toThrow(/cannot be empty/);
  });

  test('performance: average ≤ 12 ms over 10 runs', async () => {
    const runs = 10;
    const timings = [];
    for (let i = 0; i < runs; i++) {
      const start = process.hrtime.bigint();
      await hashPassword('PerfTest123!');
      const end = process.hrtime.bigint();
      timings.push(Number(end - start) / 1e6); // ms
    }
    const avg = timings.reduce((a, b) => a + b, 0) / runs;
    expect(avg).toBeLessThanOrEqual(12);
  });
});

describe('comparePassword()', () => {
  const plain = 'My$tr0ngP@ss';
  let hash;

  beforeAll(async () => {
    hash = await hashPassword(plain);
  });

  test('returns true for matching password', async () => {
    const ok = await comparePassword(plain, hash);
    expect(ok).toBe(true);
  });

  test('returns false for non‑matching password', async () => {
    const ok = await comparePassword('WrongPass', hash);
    expect(ok).toBe(false);
  });

  test('throws TypeError when hash is malformed', async () => {
    await expect(comparePassword(plain, 'not-a-bcrypt-hash')).rejects.toThrow(TypeError);
  });
});
